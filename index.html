📊 Підключення до Google Sheets

① Створіть Apps Script у вашій таблиці (один раз)
  1. Відкрийте вашу таблицю
  2. Меню Розширення → Apps Script
  3. Замініть весь код у редакторі на код нижче і натисніть Зберегти
  4. Розгорнути → Нове розгортання → Тип: Веб-застосунок
  5. Виконувати як: Я  |  Хто має доступ: Усі → Розгорнути
  6. Скопіюйте URL розгортання та вставте нижче
const TOKEN = 'crm2026secret';

function doGet(e) {
  if (!e.parameter.token || e.parameter.token !== TOKEN)
    return ContentService.createTextOutput('Unauthorized');

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const result = {};

  ['1_Реєстр ДУ', '2_Реєстр Дій'].forEach(name => {
    const sh = ss.getSheetByName(name);
    if (!sh) return;
    result[name] = sh.getDataRange().getValues().map(row =>
      row.map(v => v instanceof Date
        ? Utilities.formatDate(v, 'Europe/Kiev', 'dd.MM.yyyy')
        : String(v === null || v === undefined ? '' : v))
    );
  });

  result.ts = new Date().toISOString();
  const json = JSON.stringify(result);
  const cb = e.parameter.callback;

  // JSONP підтримка (для локального файлу)
  if (cb) {
    return ContentService
      .createTextOutput(cb + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
